  configure: function(object, configuration) {
    if (this[configuration] != null) {
      this[configuration].apply(object);
    } else {
      console.error("Configuration " + configuration + " not found!");
    }
    return object;
  },
  promisedConfigure: function(object, configuration) {
    var deferred;
    deferred = Q.defer();
    this.configure(object, configuration);
    deferred.resolve(object);
    return deferred.promise;
  },
